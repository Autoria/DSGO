## 堆排序
```go
	func HeapSort(list []int) {
		if len(list) < sz_limit {
			InsertSort(list)
		} else {
			for idx := len(list)/2 - 1; idx >= 0; idx-- {
				down(list, idx)
			}
			for sz := len(list) - 1; sz > 0; sz-- {
				list[0], list[sz] = list[sz], list[0]
				down(list[:sz], 0)
			}
		}
	}
```
